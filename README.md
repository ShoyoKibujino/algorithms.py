# чисел по возрастанию методом вставок (Insertion Sort).

def insertion_sort(arr):
    """
    Сортировка вставками.
    Временная сложность: O(n²) в худшем случае, O(n) в лучшем
    Пространственная сложность: O(1) - сортировка на месте
    """
    n = len(arr)  # Получаем длину массива
    for i in range(1, n):  # Проходим по массиву начиная со второго элемента
        key = arr[i]  # Сохраняем текущий элемент как ключ для вставки
        j = i - 1  # Начинаем сравнение с предыдущим элементом
        while j >= 0 and arr[j] > key:  # Пока не дошли до начала и предыдущий элемент больше ключа
            arr[j + 1] = arr[j]  # Сдвигаем больший элемент вправо
            j -= 1  # Переходим к следующему предыдущему элементу
        arr[j + 1] = key  # Вставляем ключ на правильное место

if __name__ == "__main__":  # Проверяем, запущен ли скрипт напрямую
    try:
        input_list_str = input().split()  # Читаем строку ввода и разбиваем по пробелам
        if not input_list_str:  # Если строка пустая
            numbers_to_sort = []  # Создаем пустой список
        else:
            numbers_to_sort = [int(x) for x in input_list_str]  # Преобразуем строки в числа
    except ValueError:  # Если произошла ошибка преобразования в число
        print("Ошибка: Введите числа, разделенные пробелами.")  # Выводим сообщение об ошибке
        numbers_to_sort = []  # Создаем пустой список

    insertion_sort(numbers_to_sort)  # Вызываем функцию сортировки
    print(*numbers_to_sort)  # Выводим отсортированный массив через пробел

# Описание. Задача 3.1. Линейный поиск числа
Напишите функцию linear_search(arr, target), которая реализует алгоритм линейного поиска. Функция должна принимать список целых чисел arr и искомое
число target. Функция возвращает индекс первого вхождения числа или -1, если число не найдено.

def linear_search(arr, target):
    for i in range(len(arr)): # Проходим по всем элементам списка
        if arr[i] == target: # Проверяем, равен ли текущий элемент искомому числу
            return i # Если нашли - сразу возвращаем индекс текущего элемента
    return -1 # Если цикл закончился и ничего не нашли - возвращаем -1

if __name__ == "__main__": # Пытаемся выполнить код, который может вызвать ошибку
    try:
        input_str = input().split() # Читаем строку ввода и разбиваем по пробелам
        if not input_str: # Проверяем, пустая ли строка
            arr = [] # Если пустая - создаем пустой список
        else:
            arr = [int(x) for x in input_str] # Если не пустая - преобразуем каждое слово в целое число
        target = int(input()) # Читаем целевое число для поиска
    except ValueError: # Если произошла ошибка при преобразовании в число
        print("Ошибка: Введите числа и целевое значение")
        arr = [] # Создаем пустой список, чтобы программа не упала
        target = 0  # Задаем значение по умолчанию
    
    print(linear_search(arr, target)) # Вызываем функцию поиска и сразу выводим результат
